// Simple debounce utility
function debounce<T extends unknown[]>(
  func: (..._args: T) => Promise<void>,
  wait: number
): (..._args: T) => void {
  let timeout: NodeJS.Timeout;
  return (..._args: T) => {
    clearTimeout(timeout);
    timeout = setTimeout(() => func(..._args), wait);
  };
}
